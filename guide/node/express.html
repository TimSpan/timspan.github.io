# express

# 基本使用

let express = require('express')
let app = express()
app.get('/api/getbook', (req, res) => {
    console.log(req.url);
    console.log(req.method);
    console.log('有人get,想要一本书');
    //send里面不能填写数字
    res.send('苏唐小堂堂')
})
app.post('/api/addbook', (req, res) => {
    res.send('添加苏唐小堂堂成功'+'请求地址:'+req.url+req.method)
    console.log('有人post,想要提交一本书');
})
//综合写法
app.use('/api/', (req, res) => {
    res.send('已经接收到你的任何api下的请求,' + '请求地址:' + req.url + req.method)
})
app.listen(8080, () => {
    console.log('express服务已启动在http://localhost:8080');
})
//http://localhost:8080/api/getbook

app.get('/api/get2/:name/:age/:sex', (req, res) => {

# express 托管静态文件

app.use(express.static('public'))

现在，你就可以访问 public 目录中的所有文件了：

app.use('/static', express.static('public'))

现在，你就可以通过带有 /static 前缀地址来访问 public 目录中的文件了。

http://localhost:3000/static/images/kitten.jpg
http://localhost:3000/static/css/style.css
http://localhost:3000/static/js/app.js
http://localhost:3000/static/images/bg.png
http://localhost:3000/static/hello.html

# 中间件

中间件函数是可以访问请求对象open in new window ( req)、响应对象open in new window( res) 和next

next()函数是 Express 路由器中的一个函数

如果当前中间件函数没有结束请求-响应循环，它必须调用next()以将控制权传递给下一个中间件函数。否则，请求将被挂起
